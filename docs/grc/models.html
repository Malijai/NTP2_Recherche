from django.db import models
from django.contrib.auth.models import User
from django.utils.translation import ugettext_lazy as _
#

Definit la liste de réponses correspondant a une variable

class Typequestion(models.Model):
    nom = models.CharField(max_length=200, )
    taille = models.CharField(max_length=200, )
# 
    def __str__(self):
        return '%s' % self.nom
#

Liste des valeurs des choix de code de violation, verdict, type de tribunal

class Listevaleur(models.Model):
    reponse_valeur = models.CharField(max_length=200)
    reponse_en = models.CharField(max_length=200, )
    reponse_fr = models.CharField(max_length=200, )
    typequestion = models.ForeignKey(Typequestion, on_delete=models.DO_NOTHING)
# 
    def __str__(self):
        return '%s' % self.reponse_en
# 
class Province(models.Model):
    reponse_en = models.CharField(max_length=30,)
    reponse_fr = models.CharField(max_length=30,)
# 
    def __str__(self):
        return '%s' % self.reponse_en
# 
class Municipalite(models.Model):
    reponse_valeur = models.CharField(max_length=20)
    reponse_en = models.CharField(max_length=200, )
    reponse_fr = models.CharField(max_length=200, )
    province = models.ForeignKey(Province, on_delete=models.DO_NOTHING)
# 
    class Meta:
        ordering = ['reponse_en']
# 
    def __str__(self):
        return '%s' % self.reponse_en
# 
class Personnegrc(models.Model):
    codeGRC = models.CharField(max_length=30)
    province = models.ForeignKey(Province, on_delete=models.DO_NOTHING)
    delit = models.IntegerField(default=1, verbose_name=_("Présence de délits?"),)
    alias = models.IntegerField(default=1, verbose_name=_("Présence d'alias?"),)
    dateprint1 = models.DateField(verbose_name=_("Ancienne date print"),)
    oldpresencefps = models.IntegerField(verbose_name=_("Présence de FPS en 2015"),)
    dateprint2 = models.DateField(verbose_name=_("Nouvelle date print"), blank=True, null=True)
    newpresencefps = models.IntegerField(verbose_name=_("Présence de FPS en 2019/20"), blank=True, null=True)
    assistant = models.ForeignKey(User, on_delete=models.DO_NOTHING)
# 
    class Meta:
        ordering = ['codeGRC']
# 
    def __str__(self):
        return '%s' % self.codeGRC

Une fiche pour chaque délit de chaque personne.

class Chezsoi(models.Model):
    personne = models.ForeignKey(Personnegrc, on_delete=models.CASCADE)
    groupedelits = models.IntegerField(default=1,)
    datesentence = models.DateField(verbose_name=_("Date"), blank=True, null=True)
    ordredelit = models.IntegerField(default=1, verbose_name=_("Ordre"),)
    codeCCdelit = models.CharField(max_length=30,)
    nombrechef = models.IntegerField(default=1,)
    amendeON = models.IntegerField(default=0,)
    detentionON = models.IntegerField(default=0,)
    probationON = models.IntegerField(default=0,)
    interdictionON = models.IntegerField(default=0,)
    surcisON = models.IntegerField(default=0,)
    typetribunal = models.ForeignKey(Listevaleur, related_name='tribunal', on_delete=models.DO_NOTHING)
    violation = models.ForeignKey(Listevaleur, related_name='violation', on_delete=models.DO_NOTHING)
    verdict = models.ForeignKey(Listevaleur, related_name='verdict', on_delete=models.DO_NOTHING)
    province = models.ForeignKey(Province, on_delete=models.DO_NOTHING)
    lieu = models.ForeignKey(Municipalite, on_delete=models.DO_NOTHING)
    assistant = models.ForeignKey(User, on_delete=models.DO_NOTHING)
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)
    card = models.IntegerField(default=1)
